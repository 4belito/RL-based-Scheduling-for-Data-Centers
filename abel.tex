% ---- Slide 0+1: Define the MDP & what changed --------------
\begin{frame}{Defining the Problem as an MDP}
  $$ \mathcal{M} = (\underbrace{\mathcal{S}}_{\text{state}},\ \underbrace{\mathcal{A}}_{\text{action}},\ \underbrace{P}_{\text{transition}},\ \underbrace{R}_{\text{reward}},\ \underbrace{\gamma}_{\text{discount}}) $$

  \vspace{0.6em}
  \begin{center}
  \renewcommand{\arraystretch}{1.25}
  \begin{tabular}{l c c}
    \toprule
    \textbf{Component} & \textbf{Earlier (v1)} & \textbf{Now (v2)} \\
    \midrule
    State      & $12$-dim aggregated   & \textcolor{SoftCoral}{\textbf{$2K{+}2$ per-job, normalised}} \\
    Action     & Discrete$(K{+}1)$ promote         & unchanged \\
    Transition & $\Delta t$ step, stochastic arrivals & unchanged \\
    Reward     & $w_u u_t - w_Q Q_t$ (unbounded)   & \textcolor{SoftCoral}{\textbf{$-w\,\tfrac{\min(D_t,n_{\text{free}})}{N}\in[-1,0]$}} \\
    Horizon    & truncate at $T$ or drained        & unchanged \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.3em}
  \centering
  \notebar{\mbox{$N$ = total nodes} \;\textbar\; \mbox{$n_{\text{free}}$ = idle nodes} \;\textbar\; \mbox{$K$ = visible queued jobs} \;\textbar\; \mbox{$D_t$ = queue demand} \;\textbar\; \mbox{$u_t$ = utilisation} \;\textbar\; \mbox{$Q_t$ = queue length} \;\textbar\; \mbox{$T$ = horizon}}
\end{frame}

% ---- Slide 2: State change (CHANGED) -----------------------
\begin{frame}{State $\;s_t$ \hfill \small\textcolor{SoftCoral}{(redesigned)}}
  \begin{columns}[T]
    \column{0.45\textwidth}
      \begin{block}{Earlier (v1): aggregated summary}
        \small
        $s_t \in \mathbb{R}^{12}$: cluster stats $+$ queue \emph{means}.
        \begin{itemize}
          \item Cluster stats: $u_t$, $n_{\text{free}}$, $Q_t$, $\dots$
          \item Queue summarised by means $\bar{q}_t,\bar{w}_t$
          \item No per-job detail
        \end{itemize}
      \end{block}

    \column{0.45\textwidth}
      \begin{block}{Now (v2): per-job features}
        \small
        $s_t \in \mathbb{R}^{2K+2}$, one entry per visible job:
        $$
          s_t = \Big[\tfrac{t}{T},\ \tfrac{n_{\text{free}}}{N},\
          \big(\tfrac{q_i}{N}\big)_{i=1}^{K},\
          \big(\tfrac{w_i}{T}\big)_{i=1}^{K}\Big]
        $$
        \begin{itemize}
          \item Exposes \emph{each} visible job, not just a mean
          \item Empty slots padded with $-1$ sentinel
        \end{itemize}
      \end{block}
  \end{columns}

  \vspace{0.4em}
  \centering
  \small\textcolor{DeepBlue}{\textbf{Why:}} per-job observation lets the policy reason about \emph{which} job to promote, not just queue averages.
  \notebar{\mbox{$u_t$ = utilisation} \;\textbar\; \mbox{$n_{\text{free}}$ = idle nodes} \;\textbar\; \mbox{$Q_t$ = queue length} \;\textbar\; \mbox{$\bar{q}_t,\bar{w}_t$ = queue means} \;\textbar\; \mbox{$q_i,w_i$ = per-job size / wait} \;\textbar\; \mbox{$t,T$ = elapsed / horizon} \;\textbar\; \mbox{$N$ = total nodes} \;\textbar\; \mbox{$K$ = visible jobs}}
\end{frame}

% ---- Slide 3: Reward change (CHANGED) ----------------------
\begin{frame}{Reward $\;R(s_t, a_t)$ \hfill \small\textcolor{SoftCoral}{(redesigned)}}
  \footnotesize
  \begin{columns}[T,onlytextwidth]
    \column{0.315\textwidth}
      \begin{block}{v1: util.\ $-$ queue}
        \begin{minipage}[t][3.5cm]{\linewidth}
        $$ r_t = w_u u_t - w_Q Q_t $$
        \begin{itemize}\setlength\itemsep{2pt}
          \item Rewards busy nodes, penalises backlog
          \item Penalty uses \textbf{raw} queue count $Q_t$
          \item \alert{Unbounded}: scales with workload size
        \end{itemize}
        \end{minipage}
      \end{block}

    \hfill
    \column{0.315\textwidth}
      \begin{block}{v2: schedulable-idle}
        \begin{minipage}[t][3.5cm]{\linewidth}
        $$ r_t = -w\,\frac{\min(D_t, n_{\text{free}})}{N} $$
        \begin{itemize}\setlength\itemsep{2pt}
          \item Charges only for idle nodes the queue \emph{could} use
          \item Bounded $[-1,0]$, \textcolor{DeepBlue}{load-agnostic}
        \end{itemize}
        \end{minipage}
      \end{block}

    \hfill
    \column{0.315\textwidth}
      \begin{block}{v2: priority slowdown}
        \begin{minipage}[t][3.5cm]{\linewidth}
        $$ r^{\text{p}}_t = -w_p\,\widehat{\textstyle\sum_{j} \tfrac{p_j}{T_j}} $$
        \begin{itemize}\setlength\itemsep{2pt}
          \item Summed over \textbf{running \& queued} jobs ($p_j$=priority, $T_j$=runtime)
          \item Bounded $[-1,0]$; favours high-priority, short jobs
        \end{itemize}
        \end{minipage}
      \end{block}
  \end{columns}

  \vspace{0.4em}
  \centering
  \small\textcolor{DeepBlue}{\textbf{Why:}} targets head-of-line blocking directly with a well-scaled signal.
  \notebar{$u_t$ = cluster utilisation \;\textbar\; $Q_t$ = raw queue length \;\textbar\; $D_t$ = total queue demand (nodes) \;\textbar\; $n_{\text{free}}$ = idle nodes \;\textbar\; $N$ = total nodes}
\end{frame}

% ---- Slide 4: Why it matters -------------------------------
\begin{frame}{Why These Changes Matter}
  \begin{columns}[T]
    \column{0.52\textwidth}
      \textbf{From v1 to v2 we gain:}
      \begin{itemize}
        \item \textbf{Bounded signals}: state in $[0,1]$, reward in $[-1,0]$
        \item \textbf{Load invariance}: no dependence on raw queue size or job distribution
        \item \textbf{Richer observation}: per-job features instead of a single mean
        \item \textbf{Direct objective}: penalise \emph{schedulable} idle capacity (head-of-line blocking)
      \end{itemize}

    \column{0.48\textwidth}
      \begin{alertblock}{Net effect}
        A policy that learns to \textbf{promote jobs that fit} the free nodes,
        draining the queue faster, and that \textbf{transfers} across
        workloads without re-tuning reward weights.
      \end{alertblock}
  \end{columns}
\end{frame}
